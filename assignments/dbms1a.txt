CREATE TABLE students (
    rno NUMBER PRIMARY KEY,
    name VARCHAR2(30) NOT NULL,
    course VARCHAR2(20),
    marks NUMBER(5),
    grade CHAR(1),
    CONSTRAINT chk_marks CHECK (marks BETWEEN 0 AND 100)
);

CREATE SEQUENCE stu_squ
START WITH 1
INCREMENT BY 1;

CREATE INDEX idx_course ON students(course);
CREATE SYNONYM stu1 FOR students;

CREATE VIEW top_view_student AS
SELECT name,course,marks
FROM STUDENTS
WHERE MARKS>85;
