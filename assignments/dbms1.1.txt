FROM world_cricket w

FROM world_cricket w
LEFT JOIN player_salary s
  ON w.name=s.name

FROM world_cricket w

  ON w.name=s.name

FROM world_cricket w
FULL OUTER JOIN player_salary s
  ON w.name=s.name
